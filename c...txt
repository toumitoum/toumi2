///for delete patient
PatientList* deletePatient(PatientList** head, int id) {
    PatientList* p = *head;
    PatientList* prev = NULL;
    
    if (p == NULL) {


printf("The list is empty\n");
        return NULL;
    }
    if (p->id == id) {
        *head = p->next;
        free(p);
        printf("Patient with ID %d is deleted\n", id);
        return *head;
    }
    while (p != NULL && p->id != id) {
        prev = p;
        p = p->next;
    }
    if (p == NULL) {
        printf("Patient with ID %d not found\n", id);
        return *head;
    }
    prev->next = p->next;
    free(p);
    printf("Patient with ID %d is deleted\n", id);
    return *head;
}
